// Timeframe-based distance (MT5 CalculateArrowDistance())
// baseDistance = 20 points, then timeframeMultiplier, then ArrowDistanceMultiplier, then _Point
// In TradingView: _Point ~ syminfo.mintick
//-------------------------
f_tfMultiplier() =>
    // MT5 mapping:
    // M1=0.5, M5=1.0, M15=1.5, M30=2.0, H1=3.0, H4=5.0, D1=8.0, W1=15.0, MN1=25.0, default=2.0
    float mult = 2.0
    int tfSeconds = timeframe.in_seconds()
    if tfSeconds < 3600  // Minutes
        if tfSeconds == 60
            mult := 0.5
        else if tfSeconds == 300
            mult := 1.0
        else if tfSeconds == 900
            mult := 1.5
        else if tfSeconds == 1800
            mult := 2.0
        else
            mult := 2.0
    else if tfSeconds < 86400  // Hours
        if tfSeconds == 3600
            mult := 3.0
        else if tfSeconds == 14400
            mult := 5.0
        else
            mult := 2.0
    else if tfSeconds == 86400  // Daily
        mult := 8.0
    else if tfSeconds == 604800  // Weekly
        mult := 15.0
    else if tfSeconds >= 2592000  // Monthly (approximately)
        mult := 25.0
    else
        mult := 2.0
    mult

f_roundToTick(price) =>
    syminfo.mintick > 0 ? math.round(price / syminfo.mintick) * syminfo.mintick : price

f_getSpreadPrice() =>
    float liveSpread = not na(ask) and not na(bid) and ask >= bid ? (ask - bid) : syminfo.mintick
    float contractValuePerTick = syminfo.mintick * syminfo.pointvalue
    bool hasValidTickValue = contractValuePerTick > 0
    hasValidTickValue ? math.max(liveSpread, syminfo.mintick) : syminfo.mintick

f_consecutiveBearish(n) =>
    bool ok = true
    for i = 1 to n
        ok := ok and (close[i] < open[i])
    ok

f_consecutiveBullish(n) =>
    bool ok = true
    for i = 1 to n
        ok := ok and (close[i] > open[i])
    ok

f_bullishEngulfingFlexible() =>
    // matches MT5 IsBullishEngulfing()
    if bar_index < minConsecutiveCandles
        false
    else
        bool consecutiveBear = f_consecutiveBearish(minConsecutiveCandles)
        bool currentBull     = close > open
        if not (consecutiveBear and currentBull)
            false
        else
            bool engulfs = false
            if useFlexibleEngulfing
                float prevSize = math.abs(close[1] - open[1])
                float curSize  = math.abs(close - open)
                float coverage = (open < close[1] and close > open[1] and prevSize != 0.0) ? math.min(curSize, prevSize) / prevSize : 0.0
                engulfs := coverage >= engulfingThreshold
            else
                engulfs := (open < close[1]) and (close > open[1])
            engulfs

f_bearishEngulfingFlexible() =>
    // matches MT5 IsBearishEngulfing()
    if bar_index < minConsecutiveCandles
        false
    else
        bool consecutiveBull = f_consecutiveBullish(minConsecutiveCandles)
        bool currentBear     = close < open
        if not (consecutiveBull and currentBear)
            false
        else
            bool engulfs = false
            if useFlexibleEngulfing
                float prevSize = math.abs(close[1] - open[1])
                float curSize  = math.abs(close - open)
                float coverage = (open > close[1] and close < open[1] and prevSize != 0.0) ? math.min(curSize, prevSize) / prevSize : 0.0
                engulfs := coverage >= engulfingThreshold
            else
                engulfs := (open > close[1]) and (close < open[1])
            engulfs

f_simpleBullishEngulfing() =>
    // matches MT5 IsSimpleBullishEngulfing()
    bar_index >= 1 and (close[1] < open[1]) and (close > open) and (open < close[1]) and (close > open[1])

f_simpleBearishEngulfing() =>
    // matches MT5 IsSimpleBearishEngulfing()
    bar_index >= 1 and (close[1] > open[1]) and (close < open) and (open > close[1]) and (close < open[1])

// Lightweight indecision filter for same-bar or next-bar READY confirmation.
f_entryConfirmBullCandle() =>
    float candleRange = high - low
    float candleBody = math.abs(close - open)
    bool rangeValid = candleRange > 0
    bool decisiveBody = rangeValid and f_safeDiv(candleBody, candleRange) >= 0.3
    close > open and decisiveBody

f_entryConfirmBearCandle() =>
    float candleRange = high - low
    float candleBody = math.abs(close - open)
    bool rangeValid = candleRange > 0
    bool decisiveBody = rangeValid and f_safeDiv(candleBody, candleRange) >= 0.3
    close < open and decisiveBody

f_strongBullCandle(atrValue) =>
    float candleRange = high - low
    float candleBody = math.abs(close - open)
    float upperWick = high - math.max(close, open)
    bool rangeValid = candleRange > 0
    bool nearHigh = rangeValid and f_safeDiv(high - close, candleRange) <= execCloseNearExtreme
    bool wickOK = candleBody > 0 and f_safeDiv(upperWick, candleBody) <= execMaxOpposingWickBody
    close > open and candleBody >= atrValue * execCandleATRMult and nearHigh and wickOK

f_strongBearCandle(atrValue) =>
    float candleRange = high - low
    float candleBody = math.abs(close - open)
    float lowerWick = math.min(close, open) - low
    bool rangeValid = candleRange > 0
    bool nearLow = rangeValid and f_safeDiv(close - low, candleRange) <= execCloseNearExtreme
    bool wickOK = candleBody > 0 and f_safeDiv(lowerWick, candleBody) <= execMaxOpposingWickBody
    close < open and candleBody >= atrValue * execCandleATRMult and nearLow and wickOK

f_htfBullBiasCalc() =>
    float htfEma = ta.ema(close, htfBiasEMALen)
    float htfLowShort = ta.lowest(low, htfBiasShortLookback)
    float htfLowLong = ta.lowest(low, htfBiasLongLookback)
    close > htfEma and htfEma >= htfEma[1] and htfLowShort > htfLowLong

f_htfBearBiasCalc() =>
    float htfEma = ta.ema(close, htfBiasEMALen)
    float htfHighShort = ta.highest(high, htfBiasShortLookback)
    float htfHighLong = ta.highest(high, htfBiasLongLookback)
    close < htfEma and htfEma <= htfEma[1] and htfHighShort < htfHighLong

f_sniperBullCandle(atrValue, requireMinorBreak) =>
    float candleRange = high - low
    float candleBody = math.abs(close - open)
    float upperWick = high - math.max(close, open)
    bool rangeValid = candleRange > 0
    bool nearHigh = rangeValid and f_safeDiv(high - close, candleRange) <= sniperCloseNearExtremePct
    bool wickOK = candleBody > 0 and f_safeDiv(upperWick, candleBody) <= sniperMaxOpposingWickPct
    bool minorBreakOK = not requireMinorBreak or high > high[1]
    bool bodyDominant = rangeValid and f_safeDiv(candleBody, candleRange) >= 0.5
    close > open and candleBody >= atrValue * sniperMinBodyATR and nearHigh and wickOK and minorBreakOK and bodyDominant

f_sniperBearCandle(atrValue, requireMinorBreak) =>
    float candleRange = high - low
    float candleBody = math.abs(close - open)
    float lowerWick = math.min(close, open) - low
    bool rangeValid = candleRange > 0
    bool nearLow = rangeValid and f_safeDiv(close - low, candleRange) <= sniperCloseNearExtremePct
    bool wickOK = candleBody > 0 and f_safeDiv(lowerWick, candleBody) <= sniperMaxOpposingWickPct
    bool minorBreakOK = not requireMinorBreak or low < low[1]
    bool bodyDominant = rangeValid and f_safeDiv(candleBody, candleRange) >= 0.5
    close < open and candleBody >= atrValue * sniperMinBodyATR and nearLow and wickOK and minorBreakOK and bodyDominant

f_bodySize(idx) =>
    math.abs(close[idx] - open[idx])

f_upperWick(idx) =>
    high[idx] - math.max(close[idx], open[idx])

f_lowerWick(idx) =>
    math.min(close[idx], open[idx]) - low[idx]

f_hammer() =>
    if bar_index < 1
        false
    else
        float body = f_bodySize(0)
        float upperWick = f_upperWick(0)
        float lowerWick = f_lowerWick(0)
        float barRange = high - low
        barRange > 0 and lowerWick >= body * 2.0 and upperWick <= body * 0.5 and close > open

f_shootingStar() =>
    if bar_index < 1
        false
    else
        float body = f_bodySize(0)
        float upperWick = f_upperWick(0)
        float lowerWick = f_lowerWick(0)
        float barRange = high - low
        barRange > 0 and upperWick >= body * 2.0 and lowerWick <= body * 0.5 and close < open

f_morningStar() =>
    if bar_index < 2
        false
    else
        bool firstBear = close[2] < open[2]
        float firstBody = f_bodySize(2)
        bool secondSmall = f_bodySize(1) <= firstBody * 0.3
        bool thirdBull = close > open
        bool thirdEngulfs = close > (open[2] + close[2]) / 2.0
        firstBear and secondSmall and thirdBull and thirdEngulfs

f_eveningStar() =>
    if bar_index < 2
        false
    else
        bool firstBull = close[2] > open[2]
        float firstBody = f_bodySize(2)
        bool secondSmall = f_bodySize(1) <= firstBody * 0.3
        bool thirdBear = close < open
        bool thirdEngulfs = close < (open[2] + close[2]) / 2.0
        firstBull and secondSmall and thirdBear and thirdEngulfs

f_insideBarBullish() =>
    if bar_index < 1
        false
    else
        bool inside = high < high[1] and low > low[1]
        bool bullish = close > open
        inside and bullish

f_insideBarBearish() =>
    if bar_index < 1
        false
    else
        bool inside = high < high[1] and low > low[1]
        bool bearish = close < open
        inside and bearish

f_risingThreeMethods() =>
    if bar_index < 4
        false
    else
        bool firstBull = close[4] > open[4]
        float firstBody = f_bodySize(4)
        bool secondBear = close[3] < open[3] and high[3] < high[4] and low[3] > low[4]
        bool thirdBear = close[2] < open[2] and high[2] < high[4] and low[2] > low[4]
        bool fourthBear = close[1] < open[1] and high[1] < high[4] and low[1] > low[4]
        bool fifthBull = close > open and close > close[4]
        firstBull and secondBear and thirdBear and fourthBear and fifthBull

f_fallingThreeMethods() =>
    if bar_index < 4
        false
    else
        bool firstBear = close[4] < open[4]
        float firstBody = f_bodySize(4)
        bool secondBull = close[3] > open[3] and high[3] < high[4] and low[3] > low[4]
        bool thirdBull = close[2] > open[2] and high[2] < high[4] and low[2] > low[4]
        bool fourthBull = close[1] > open[1] and high[1] < high[4] and low[1] > low[4]
        bool fifthBear = close < open and close < close[4]
        firstBear and secondBull and thirdBull and fourthBull and fifthBear

// Check if zone is strong at pivot candle
f_isStrongZone(pivotBarIndex) =>
    bool isStrong = false
    if pivotBarIndex >= 0
        float pivotRange = high[pivotBarIndex] - low[pivotBarIndex]
        float pivotVol = volume[pivotBarIndex]
        
        if useRangeStrength and pivotRange > atr[pivotBarIndex] * atrMultStrong
            isStrong := true
        else if useVolumeStrength and pivotVol > volSMA[pivotBarIndex] * volMultStrong
            isStrong := true
    isStrong

// Prune old zones based on lookback window
f_pruneOldZones() =>
    int cutoff = bar_index - zoneLookbackBars
    
    // Prune supply zones (oldest first, index 0)
    bool supplyDone = false
    while array.size(supplyBarIndex) > 0 and not supplyDone
        int oldestBarIdx = array.get(supplyBarIndex, 0)
        if oldestBarIdx < cutoff
            float _1 = array.shift(supplyTopPrices)
            float _2 = array.shift(supplyBottomPrices)
            bool _3 = array.shift(supplyIsStrong)
            int _4 = array.shift(supplyBarIndex)
            int _5 = array.shift(supplyTouchCount)
        else
            supplyDone := true
    
    // Prune demand zones (oldest first, index 0)
    bool demandDone = false
    while array.size(demandBarIndex) > 0 and not demandDone
        int oldestBarIdx = array.get(demandBarIndex, 0)
        if oldestBarIdx < cutoff
            float _1 = array.shift(demandTopPrices)
            float _2 = array.shift(demandBottomPrices)
            bool _3 = array.shift(demandIsStrong)
            int _4 = array.shift(demandBarIndex)
            int _5 = array.shift(demandTouchCount)
        else
            demandDone := true
    
    0

// Create supply zone (stores prices only, no visible lines)
f_createSupplyZone(pivotPrice, pivotBarIdx, isStrong) =>
    if not na(pivotPrice)
        if array.size(supplyTopPrices) >= maxSupplyZones
            // Remove oldest zone
            if array.size(supplyTopPrices) > 0
                array.shift(supplyTopPrices)
                array.shift(supplyBottomPrices)
                array.shift(supplyIsStrong)
                array.shift(supplyBarIndex)
                array.shift(supplyTouchCount)
        
        float topPrice = pivotPrice
        float bottomPrice = pivotPrice - atr[pivotBarIdx] * zoneATRMult
        
        array.push(supplyTopPrices, topPrice)
        array.push(supplyBottomPrices, bottomPrice)
        array.push(supplyIsStrong, isStrong)
        array.push(supplyBarIndex, pivotBarIdx)
        array.push(supplyTouchCount, 0)
        
        // Prune old zones after adding new one
        f_pruneOldZones()

// Create demand zone (stores prices only, no visible lines)
f_createDemandZone(pivotPrice, pivotBarIdx, isStrong) =>
    if not na(pivotPrice)
        if array.size(demandTopPrices) >= maxDemandZones
            // Remove oldest zone
            if array.size(demandTopPrices) > 0
                array.shift(demandTopPrices)
                array.shift(demandBottomPrices)
                array.shift(demandIsStrong)
                array.shift(demandBarIndex)
                array.shift(demandTouchCount)
        
        float bottomPrice = pivotPrice
        float topPrice = pivotPrice + atr[pivotBarIdx] * zoneATRMult
        
        array.push(demandTopPrices, topPrice)
        array.push(demandBottomPrices, bottomPrice)
        array.push(demandIsStrong, isStrong)
        array.push(demandBarIndex, pivotBarIdx)
        array.push(demandTouchCount, 0)
        
        // Prune old zones after adding new one
        f_pruneOldZones()

    int zoneCutoff = bar_index - zoneLookbackBars
    if array.size(supplyTopPrices) > 0

// Helper: Check if price is inside an ALLOWED supply zone (filtered by zone type)

    int zoneCutoff = bar_index - zoneLookbackBars
    if array.size(supplyTopPrices) > 0
        for i = 0 to array.size(supplyTopPrices) - 1
            // Skip zones older than cutoff

            if array.size(supplyBarIndex) > i
                int zoneBarIdx = array.get(supplyBarIndex, i)
                if zoneBarIdx < zoneCutoff
                    zoneValid := false
            if zoneValid
                float top = array.get(supplyTopPrices, i)
                float bot = array.get(supplyBottomPrices, i)
                if not na(top) and not na(bot)
                    if px <= top and px >= bot
                        // Determine zone type
                        bool isHot = (i == nearestSupplyIdx)
                        bool isStrong = false
                        if array.size(supplyIsStrong) > i
                            isStrong := array.get(supplyIsStrong, i)
                        bool isNormal = not isStrong and not isHot
                        
                        // Check if this zone type is allowed
                        bool allowed = false
                        if isHot and allowHotZones
                            allowed := true
                        else if isStrong and allowStrongZones
                            allowed := true
                        else if isNormal and allowNormalZones
                            allowed := true
                        
                        if allowed
                            inside := true
                            break
    inside

            if array.size(supplyBarIndex) > i

                if zoneBarIdx < zoneCutoff
                    zoneValid := false
            if zoneValid
                float top = array.get(supplyTopPrices, i)
                float bot = array.get(supplyBottomPrices, i)
                if not na(top) and not na(bot)
                    if px <= top and px >= bot
                        // Determine zone type
                        bool isHot = (i == nearestSupplyIdx)
                        bool isStrong = false
                        if array.size(supplyIsStrong) > i
                            isStrong := array.get(supplyIsStrong, i)
                        bool isNormal = not isStrong and not isHot
                        
                        // Check if this zone type is allowed
                        bool allowed = false
                        if isHot and allowHotZones
                            allowed := true
                        else if isStrong and allowStrongZones
                            allowed := true
                        else if isNormal and allowNormalZones
                            allowed := true
                        
                        if allowed
                            result := top
                            break
    result
